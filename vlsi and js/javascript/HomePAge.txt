<!DOCTYPE html>
<html xmlns="http://www.w3.org/1999/xhtml">
<head>
    <title>Home Page Background Change</title>

    <script type="text/javascript">
        function changeColorMouse() {
            document.body.style.backgroundColor = "lightblue";
        }

        function resetColorMouse() {
            document.body.style.backgroundColor = "white";
        }

        function changeColorFocus() {
            document.body.style.backgroundColor = "lightgreen";
        }

        function resetColorFocus() {
            document.body.style.backgroundColor = "white";
        }
    </script>

</head>
<body>
    <h2>Welcome to My Home Page</h2>

    <!-- Mouse Over -->
    <p onmouseover="changeColorMouse()" onmouseout="resetColorMouse()">
        Hover on this text to change background color (Mouse Over)
    </p>

    <!-- On Focus -->
    <input type="text" onfocus="changeColorFocus()" onblur="resetColorFocus()" 
           placeholder="Click here to trigger focus event"/>
</body>
</html>
