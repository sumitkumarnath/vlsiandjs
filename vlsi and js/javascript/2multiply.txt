<script type="text/javascript">
// <![CDATA[
function generateTable() {
    var n = parseInt(document.getElementById("num").value);

    if (isNaN(n)) {
        alert("Enter a valid number!");
        return;
    }

    var result = "";
    for (var i = 1; i <= 10; i++) {
        result += n + " Ã— " + i + " = " + (n * i) + "\n";
    }

    document.getElementById("output").textContent = result;
    console.log(result);
}
