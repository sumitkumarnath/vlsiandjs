    <label>Height: <input id="triHeight" type="number" step="any" min="0" value="5" /></label>
    <button type="button" onclick="calcTriangle()">Calculate Triangle Area</button>

    <label>Width: <input id="rectWid" type="number" step="any" min="0" value="4" /></label>

  <div class="box" id="circleBox">
    <h3>Circle (area = π × r²)</h3>

  <script type="text/javascript">
  // <![CDATA[
    /**
     * Utility: format number to max 6 decimal places (trims trailing zeros)
     * returns string
     */
    function formatNumber(n) {
      if (!isFinite(n)) return String(n);
      return parseFloat(n.toFixed(6)).toString();
    }

    // Triangle: 0.5 * base * height
    function areaTriangle(base, height) {
      return 0.5 * base * height;
    }

    // Rectangle: length * width
    function areaRectangle(length, width) {
      return length * width;
    }

    // Circle: Math.PI * r * r
    function areaCircle(radius) {
      return Math.PI * radius * radius;
    }

    // Handlers called from buttons: read inputs, compute, show result and log to console
    function calcTriangle() {
      var base = parseFloat(document.getElementById('triBase').value) || 0;
      var height = parseFloat(document.getElementById('triHeight').value) || 0;

      var area = areaTriangle(base, height);
      var out = 'Triangle area (0.5 × ' + base + ' × ' + height + ') = ' + formatNumber(area);
      document.getElementById('triResult').textContent = 'Result: ' + out;
      console.log(out);
    }

    function calcRectangle() {
      var length = parseFloat(document.getElementById('rectLen').value) || 0;
      var width  = parseFloat(document.getElementById('rectWid').value) || 0;

      var area = areaRectangle(length, width);
      var out = 'Rectangle area (' + length + ' × ' + width + ') = ' + formatNumber(area);
      document.getElementById('rectResult').textContent = 'Result: ' + out;
      console.log(out);
    }

    function calcCircle() {
      var r = parseFloat(document.getElementById('cirRadius').value) || 0;

      var area = areaCircle(r);
      var out = 'Circle area (π × ' + r + '²) = ' + formatNumber(area);
      document.getElementById('cirResult').textContent = 'Result: ' + out;
      console.log(out);
    }

    // Make functions available on window so you can call them directly from console if needed:
    window.areaTriangle = areaTriangle;
    window.areaRectangle = areaRectangle;
    window.areaCircle = areaCircle;
    window.calcTriangle = calcTriangle;
    window.calcRectangle = calcRectangle;
    window.calcCircle = calcCircle;
  // ]]>
  </script>
