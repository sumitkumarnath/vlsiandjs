<!DOCTYPE html>
<html xmlns="http://www.w3.org/1999/xhtml">
<head>
    <title>Simple Calculator</title>

    <script type="text/javascript">
        function calculate(op) {
            var n1 = document.getElementById("num1").value;
            var n2 = document.getElementById("num2").value;

            if (n1 === "" || (op !== 'sq' && n2 === "")) {
                alert("Please enter valid inputs!");
                return;
            }

            n1 = parseFloat(n1);
            n2 = parseFloat(n2);

            var result;

            if (op === '+') result = n1 + n2;
            else if (op === '-') result = n1 - n2;
            else if (op === '*') result = n1 * n2;
            else if (op === '/') {
                if (n2 === 0) { alert("Cannot divide by zero!"); return; }
                result = n1 / n2;
            }
            else if (op === 'sq') result = n1 * n1;

            document.getElementById("result").value = result;
        }
    </script>

</head>
<body>
    <h2>Simple Calculator</h2>

    <input type="text" id="num1" placeholder="Enter Number 1"/>
    <input type="text" id="num2" placeholder="Enter Number 2"/>

    <br/><br/>

    <button onclick="calculate('+')">Add</button>
    <button onclick="calculate('-')">Subtract</button>
    <button onclick="calculate('*')">Multiply</button>
    <button onclick="calculate('/')">Divide</button>
    <button onclick="calculate('sq')">Square (Num1)</button>

    <br/><br/>

    Output: <input type="text" id="result" readonly/>
</body>
</html>
