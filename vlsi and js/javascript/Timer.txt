<script type="text/javascript">
// <![CDATA[
var timer;

function startCountdown() {
    var time = parseInt(document.getElementById("seconds").value);

    if (isNaN(time) || time <= 0) {
        alert("Enter valid seconds");
        return;
    }

    document.getElementById("display").textContent = "Time: " + time;

    clearInterval(timer); // reset old timer if running

    timer = setInterval(function () {
        time--;
        document.getElementById("display").textContent = "Time: " + time;

        if (time <= 0) {
            clearInterval(timer);
            document.getElementById("display").textContent = "Time Up!";
            alert("Countdown Finished!");
        }
    }, 1000);
}
// ]]>
</script>
